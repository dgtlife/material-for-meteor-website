// To switch to the 'standard' mode, use the following commands, in sequence,
// in the browser console.
> MD.dqS('[data-header-panel]').removeAttribute('data-mode')
> MD.initializeHeaderPanelSystem()

// To switch to the 'scroll' mode, use the following commands, in sequence,
// in the browser console.
> MD.dqS('[data-header-panel]').setAttribute('data-mode', 'scroll')
> MD.initializeHeaderPanelSystem()

// To switch to the 'waterfall' mode, use the following commands, in sequence,
// in the browser console.
> MD.dqS('[data-header-panel]').setAttribute('data-mode', 'waterfall')
> MD.initializeHeaderPanelSystem()

// To switch to the 'waterfall-collapse' mode, use the following commands, in
// sequence, in the browser console.
> MD.dqS('[data-header-panel]').setAttribute('data-mode', 'waterfall-collapse')
> MD.initializeHeaderPanelSystem()

// In order to experience the effect of 'expand-on-scroll' in
// 'waterfall-collapse' mode, after you have experienced the default behavior
// of 'waterfall-collapse' mode, use the following commands, in sequence, in the
// browser console, while in 'water-fall-collapse' mode.
> MD.dqS('[data-header-panel]').setAttribute('data-expand-on-scroll', 'true')
> MD.initializeHeaderPanelSystem()

// When switching away from the 'expand-on-scroll' sub-mode be sure to remove
// its corresponding attribute with the following command.
> MD.dqS('[data-header-panel]').removeAttribute('data-expand-on-scroll')

// To switch to the 'cover' mode, use the following commands, in sequence, in
// the browser console.
> MD.dqS('[data-header-panel]').setAttribute('data-mode', 'cover')
> MD.initializeHeaderPanelSystem()

// To switch back to the 'seamed' mode, use the following commands, in sequence,
// in the browser console.
> MD.dqS('[data-header-panel]').setAttribute('data-mode', 'seamed')
> MD.initializeHeaderPanelSystem()
